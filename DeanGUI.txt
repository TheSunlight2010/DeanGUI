-- helper to find first humanoid in camera forward direction within range
-- First attempts a raycast; if that misses (or hits a non-character part) it falls back to a viewport-proximity search
local function findHumanoidInView(maxDistance)
	maxDistance = maxDistance or 120
	if not camera then return nil end

	-- Raycast attempt (exclude player's character)
	local origin = camera.CFrame.Position
	local dir = camera.CFrame.LookVector.Unit * maxDistance
	local params = RaycastParams.new()
	params.FilterDescendantsInstances = {player.Character}
	params.FilterType = Enum.RaycastFilterType.Blacklist
	params.IgnoreWater = true
	local result = workspace:Raycast(origin, dir, params)
	if result and result.Instance then
		local part = result.Instance
		local model = part:FindFirstAncestorWhichIsA("Model")
		if model then
			local humanoid = model:FindFirstChildOfClass("Humanoid")
			if humanoid and humanoid.Health ~= nil then
				return humanoid, model
			end
		end
	end

	-- Fallback: viewport proximity search around center of screen (covers aiming at any visible part)
	local viewportSize = camera.ViewportSize
	if not viewportSize or viewportSize.X == 0 then
		return nil, nil
	end
	local centerX = viewportSize.X * 0.5
	local centerY = viewportSize.Y * 0.5
	local maxScreenRadius = math.min(viewportSize.X, viewportSize.Y) * 0.08 -- tolerance radius (~8% of smaller dim)

	local bestDist = math.huge
	local bestHum, bestModel = nil, nil
	for _, inst in pairs(workspace:GetDescendants()) do
		if inst and inst:IsA("Model") then
			local humanoid = inst:FindFirstChildOfClass("Humanoid")
			if humanoid and humanoid.Health ~= nil then
				-- find any visible BasePart to project
				local part = inst:FindFirstChildWhichIsA("BasePart")
				if part and part.Parent then
					local vp, onScreen = camera:WorldToViewportPoint(part.Position)
					if onScreen and vp.Z > 0 then
						local dx = vp.X - centerX
						local dy = vp.Y - centerY
						local dist = math.sqrt(dx*dx + dy*dy)
						-- ensure within distance range
						if dist <= maxScreenRadius and (part.Position - origin).Magnitude <= maxDistance then
							if dist < bestDist then
								bestDist = dist
								bestHum = humanoid
								bestModel = inst
							end
						end
					end
				end
			end
		end
	end

	if bestHum and bestModel then
		return bestHum, bestModel
	end

	return nil, nil
end

-- Cleanse effect: applies white highlight and temporary "Cleansed!" label above target for 3s
local function applyCleanseTo(humanoidModel)
	if not humanoidModel or not humanoidModel.PrimaryPart then
		-- attempt to pick any part as adornee
		local anyPart = humanoidModel and humanoidModel:FindFirstChildWhichIsA("BasePart")
		if anyPart then
			humanoidModel.PrimaryPart = anyPart
		else
			return
		end
	end

	-- Highlight (uses Roblox Highlight instance)
	local highlight = Instance.new("Highlight")
	highlight.FillColor = Color3.fromRGB(255,255,255)
	highlight.OutlineColor = Color3.fromRGB(240,240,240)
	highlight.Name = "CleanseHighlight"
	highlight.Adornee = humanoidModel
	highlight.Parent = humanoidModel

	-- Billboard "[Name] has been cleansed!"
	local tagGui = Instance.new("BillboardGui")
	-- larger billboard to better showcase bigger text
	tagGui.Size = UDim2.new(0,200,0,56)
	tagGui.AlwaysOnTop = true
	tagGui.StudsOffset = Vector3.new(0, (humanoidModel:GetExtentsSize().Y or 4) + 1, 0)
	tagGui.Parent = humanoidModel
	local tagLabel = Instance.new("TextLabel")
	tagLabel.Size = UDim2.new(1,0,1,0)
	tagLabel.BackgroundTransparency = 1
	tagLabel.Font = Enum.Font.GothamBold
	-- increase base text size while keeping TextScaled to ensure good fit
	tagLabel.TextSize = 28
	tagLabel.TextColor3 = Color3.fromRGB(255,255,255)
	local displayName = (humanoidModel.Name and tostring(humanoidModel.Name)) or "Target"
	tagLabel.Text = displayName .. " has been cleansed!"
	tagLabel.TextScaled = true
	tagLabel.Parent = tagGui

	-- cleanup after 3 seconds
	spawn(function()
		wait(3)
		pcall(function() if highlight and highlight.Parent then highlight:Destroy() end end)
		pcall(function() if tagGui and tagGui.Parent then tagGui:Destroy() end end)
	end)
end

-- cooldown handling
local cleanseCooldown = 10
local lastCleanseAt = 0

cleanseButton.MouseButton1Click:Connect(function()
	local now = tick()
	if now - lastCleanseAt < cleanseCooldown then
		-- still cooling down; brief flash
		local prev = cleanseButton.BackgroundColor3
		cleanseButton.BackgroundColor3 = Color3.fromRGB(80,80,80)
		task.delay(0.25, function()
			if cleanseButton and cleanseButton.Parent then
				cleanseButton.BackgroundColor3 = prev
			end
		end)
		return
	end

	-- begin cleanse sequence
	lastCleanseAt = now
	-- visual disable and cooldown tint
	cleanseButton.Active = false
	cleanseButton.Text = "COOLDOWN"
	cleanseButton.BackgroundColor3 = Color3.fromRGB(80,80,80)

	-- show crosshair and countdown
	cleanseCrosshair.Visible = true
	for i=3,1,-1 do
		countdownLabel.Text = "AIMING IN " .. i .. "..."
		wait(1)
	end
	countdownLabel.Text = ""

	-- perform detection in camera forward direction
	local humanoid, model = findHumanoidInView(120)
	if humanoid and model then
		applyCleanseTo(model)
	end

	-- hide crosshair
	cleanseCrosshair.Visible = false

	-- start cooldown timer (visual progress on text)
	spawn(function()
		local finish = lastCleanseAt + cleanseCooldown
		while tick() < finish do
			local rem = math.ceil(finish - tick())
			if cleanseButton and cleanseButton.Parent then
				cleanseButton.Text = "COOLDOWN ("..rem..")"
			end
			wait(1)
		end
		-- restore
		if cleanseButton and cleanseButton.Parent then
			cleanseButton.Text = "CLEANSE"
			cleanseButton.BackgroundColor3 = Color3.fromRGB(40,40,40)
			cleanseButton.Active = true
		end
	end)
end)

-- Progressive spawn-chance and runtime handling for "The Code"
local codeBaseDenom = 12        -- denominator (12)
local codeDenom = codeBaseDenom -- kept for clarity (always 12 here)
local codeNumerator = 1         -- starts as 1 in 12
local codeSpawnInterval = 5     -- seconds between attempts
local codeInstance = false
local codeSpawnedAt = 0
local codeDespawnTimeout = 5    -- seconds until kick if not rebooted

-- attempt spawn loop
spawn(function()
	while true do
		task.wait(codeSpawnInterval)

		-- if infestation already present check timeout
		if codeInstance then
			if tick() - codeSpawnedAt >= codeDespawnTimeout then
				-- cleanup then kick
				stopInfestation()
				codeInstance = false
				-- Kick message for The Code
				local kickText = "DIED TO: THE CODE \n HOW TO COUNTER: REBOOT. \n NO MESSAGE PROVIDED."
				task.wait(0.1)
				player:Kick(kickText)
				break
			end
		else
			-- roll with progressive chance: success if random <= numerator
			local roll = math.random(1, codeDenom)
			if roll <= codeNumerator then
				-- spawn The Code infestation
				startInfestation()
				codeInstance = true
				codeSpawnedAt = tick()
				-- reset chance back to 1 in 12 for next cycle
				codeNumerator = 1
			else
				-- increase chance by 1 in 12 for next attempt, up to 12 in 12
				codeNumerator = math.min(codeDenom, codeNumerator + 1)
			end
		end
	end
end)

	task.delay(0.35, function()

			rebootButton.BackgroundColor3 = prevColor
		end
	end)
end)

-- Helper: create rabbit model behind player
local function createRabbit()
	-- ensure previous cleared
	if rabbitInstance and rabbitInstance.Parent then
		rabbitInstance:Destroy()
		rabbitInstance = nil
	end

	local character = player.Character
	if not character then return end
	local hrp = character:FindFirstChild("HumanoidRootPart") or character:FindFirstChild("Torso") or character:FindFirstChild("Humanoid") and character.HumanoidRootPart
	if not hrp then return end

	-- Position behind player roughly 5 studs
	local behindPos = hrp.Position - (camera.CFrame.LookVector.Unit * 5)

	local model = Instance.new("Model")
	model.Name = RABBIT_NAME

	local body = Instance.new("Part")
	body.Name = "Body"
	body.Size = Vector3.new(hrp.Size.X, hrp.Size.Y, hrp.Size.Z) -- match player size roughly
	body.Position = behindPos
	body.Anchored = true -- keep rabbit stationary once placed
	body.CanCollide = false
	body.Transparency = 0
	body.BrickColor = BrickColor.new("Really black")
	body.Parent = model

	local mesh = Instance.new("CylinderMesh")
	mesh.Parent = body

	-- heavy black smoke covering body (dense so base is barely visible)
	local smoke = Instance.new("ParticleEmitter")
	smoke.Name = "RabbitSmoke"
	smoke.Color = ColorSequence.new(Color3.new(0,0,0))
	smoke.LightEmission = 0
	-- larger, denser particles
	smoke.Size = NumberSequence.new({NumberSequenceKeypoint.new(0, 2), NumberSequenceKeypoint.new(1, 4)})
	smoke.Rate = 300
	smoke.Lifetime = NumberRange.new(1,2)
	smoke.Speed = NumberRange.new(0.1,0.5)
	smoke.Transparency = NumberSequence.new({NumberSequenceKeypoint.new(0, 0.2), NumberSequenceKeypoint.new(1, 0.8)})
	smoke.LockedToPart = true
	smoke.Parent = body

	-- label above
	local billboard = Instance.new("BillboardGui")
	billboard.Size = UDim2.new(0,200,0,50)
	billboard.StudsOffset = Vector3.new(0, hrp.Size.Y + 1, 0)
	billboard.AlwaysOnTop = true
	billboard.Parent = model

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(1,0,1,0)
	label.BackgroundTransparency = 1
	label.Text = "THE RABBIT"
	label.TextColor3 = Color3.fromRGB(100,170,255) -- blue text
	label.Font = Enum.Font.GothamBold
	label.TextScaled = true
	label.Parent = billboard

	model.Parent = workspace

	-- store reference
	rabbitInstance = model
	rabbitSpawnedAt = tick()
	gazeAccum = 0

	-- keep rabbit Y position synced to player Y
	if rabbitYConn then
		rabbitYConn:Disconnect()
		rabbitYConn = nil
	end
	rabbitYConn = RunService.RenderStepped:Connect(function()
		local char = player.Character
		local hrpLocal = char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso"))
		if rabbitInstance and rabbitInstance.Parent and hrpLocal then
			local b = rabbitInstance:FindFirstChild("Body")
			if b then
				local p = b.Position
				b.Position = Vector3.new(p.X, hrpLocal.Position.Y, p.Z)
			end
		end
	end)

	-- Freeze the player: store previous values and zero movement
	local humanoid = character:FindFirstChildOfClass("Humanoid")
	-- default to not touching player movement unless we determine they were free to move
	rabbitDidFreeze = false
	if humanoid then
		-- detect whether the player currently can move / is not anchored
		local hrpLocal = character:FindFirstChild("HumanoidRootPart") or character:FindFirstChild("Torso")
		local playerCanMove = (humanoid.WalkSpeed > 0 or (humanoid.JumpPower and humanoid.JumpPower > 0)) and not (hrpLocal and hrpLocal.Anchored)

		if playerCanMove then
			-- save previous player movement values into prevPlayerState so we can restore reliably later
			prevPlayerState.PrevWalkSpeed = humanoid.WalkSpeed
			prevPlayerState.PrevJumpPower = humanoid.JumpPower or 50
			-- freeze movement
			pcall(function() humanoid.WalkSpeed = 0 end)
			pcall(function() humanoid.JumpPower = 0 end)

			-- anchor the HumanoidRootPart to prevent physics movement while preserving posture.
			if hrpLocal then
				-- store previous anchored state so we can restore it later
				prevPlayerState.PrevHRPAnchored = hrpLocal.Anchored
				pcall(function() hrpLocal.Anchored = true end)
			end

			rabbitDidFreeze = true
		else
			-- player already couldn't move; ensure we won't attempt to restore anything later
			prevPlayerState = {}
			rabbitDidFreeze = false
		end
	end

	-- Make rabbit visible and add a blue outline; also switch the screen overlay to green while spawned
	local function updateVisibility()
		if not rabbitInstance then return end
		for _, part in ipairs(rabbitInstance:GetDescendants()) do
			if part:IsA("BasePart") then
				pcall(function() part.LocalTransparencyModifier = 0 end)
				part.Transparency = 0
			elseif part:IsA("BillboardGui") then
				part.Enabled = true
			elseif part:IsA("ParticleEmitter") then
				part.Enabled = true
			end
		end

		-- (outline removed) keep rabbit visibility handled by particle and overlay state

		-- change overlay to green to indicate spawn
		if rabbitOverlay then
			rabbitOverlay.BackgroundColor3 = Color3.fromRGB(30,150,30)
			rabbitOverlay.BackgroundTransparency = 0.85
		end
	end

	updateVisibility()

	-- rabbit stays stationary; do not continuously reposition it
end

-- helper to reliably restore player movement and HRP anchored state
local function restorePlayerMovement()
	local character = player.Character
	if not character then
		prevPlayerState = {}
		return
	end
	local humanoid = character:FindFirstChildOfClass("Humanoid")
	if humanoid then
		if prevPlayerState.PrevWalkSpeed ~= nil then
			pcall(function() humanoid.WalkSpeed = prevPlayerState.PrevWalkSpeed end)
		end
		if prevPlayerState.PrevJumpPower ~= nil then
			pcall(function() humanoid.JumpPower = prevPlayerState.PrevJumpPower end)
		end
	end
	local hrpLocal = character:FindFirstChild("HumanoidRootPart") or character:FindFirstChild("Torso")
	if hrpLocal then
		if prevPlayerState.PrevHRPAnchored ~= nil then
			pcall(function() hrpLocal.Anchored = prevPlayerState.PrevHRPAnchored end)
		else
			pcall(function() hrpLocal.Anchored = false end)
		end
	end
	-- clear stored state after restoring
	prevPlayerState = {}
end

local function destroyRabbit()
	-- restore player movement first (use safe helper)
	restorePlayerMovement()

	if rabbitInstance and rabbitInstance.Parent then
		-- remove outline if present
		local outline = rabbitInstance:FindFirstChild("RabbitOutline")
		if outline then outline:Destroy() end
		rabbitInstance:Destroy()
	end
	-- disconnect Y-sync connection if present
	if rabbitYConn then
		pcall(function() rabbitYConn:Disconnect() end)
		rabbitYConn = nil
	end

	rabbitInstance = nil
	-- reset overlay to red
	if rabbitOverlay then
		rabbitOverlay.BackgroundColor3 = Color3.fromRGB(150,30,30)
		rabbitOverlay.BackgroundTransparency = 0.9
	end
end

		-- if there is no rabbit but prevPlayerState still holds values, force restore
		if not (rabbitInstance and rabbitInstance.Parent) and next(prevPlayerState) ~= nil then
			restorePlayerMovement()
		end
	end
end)

-- spawn chance logic
spawn(function()
	while true do
		task.wait(spawnInterval)
		-- Skip spawn attempts entirely if the player can't control the camera (cutscene/forced camera)
		if camera and camera.CameraType ~= Enum.CameraType.Custom then
			-- gently keep chance progression unchanged (so it still approaches spawn when control returns)
			currentDenom = math.max(2, currentDenom - 1)
			continue
		end

		-- if already spawned skip
		if rabbitInstance then
			-- check timeout for automatic kill
			if tick() - rabbitSpawnedAt >= despawnTimeout then
				-- kick player with random message
				local msg = randomMessages[math.random(1,#randomMessages)]
				if msg:find("%%s") then
					msg = string.format(msg, player.Name)
				end
				local kickText = "DIED TO: THE RABBIT. \n HOW TO COUNTER: USE X-RAY TO STARE AT HER. \n MESSAGE: " .. msg
				-- cleanup then kick
				destroyRabbit()
				task.wait(0.1)
				player:Kick(kickText)
				break
			end
			-- continue loop while rabbit exists
		else
			-- roll spawn: success if math.random(1,currentDenom) == 1
			local roll = math.random(1, currentDenom)
			if roll == 1 then
				-- do not spawn rabbit if Hesitation is active
				if hesitationInstance and hesitationInstance.Parent then
					-- skip this spawn attempt; reset denom so chance increases normally next tick
					currentDenom = math.max(2, currentDenom - 1)
				else
					-- spawn rabbit behind player
					createRabbit()
					-- reset denom
					currentDenom = baseChanceDenom
					-- ensure overlay shows green to indicate rabbit
					if rabbitOverlay then
						rabbitOverlay.BackgroundColor3 = Color3.fromRGB(30,150,30)
						rabbitOverlay.BackgroundTransparency = 0.85
					end
				end
			else
				-- increase chance (decrease denom) but not below 2
				currentDenom = math.max(2, currentDenom - 1)
			end
		end
	end
end)

-- Helper: test if Hesitation eye label is currently on-screen
local function isHesitationEyeOnScreen()
	if not hesitationInstance or not hesitationInstance.Parent then
		return false
	end
	local pillarPart = hesitationInstance:FindFirstChild("Pillar")
	local bb = pillarPart and pillarPart:FindFirstChildOfClass("BillboardGui") or nil
	if not pillarPart or not bb or not camera then
		return false
	end
	local worldPos = pillarPart.Position + (bb.StudsOffset or Vector3.new(0,0,0))
	local vpPos, onScreen = camera:WorldToViewportPoint(worldPos)
	local inView = onScreen and vpPos.Z > 0 and vpPos.X >= 0 and vpPos.Y >= 0 and vpPos.X <= camera.ViewportSize.X and vpPos.Y <= camera.ViewportSize.Y
	return inView
end

local function createHesitation()
	-- ensure previous cleared
	if hesitationInstance and hesitationInstance.Parent then
		hesitationInstance:Destroy()
		hesitationInstance = nil
	end

	if not character then return end
	local hrp = character:FindFirstChild("HumanoidRootPart") or character:FindFirstChild("Torso")
	if not hrp then return end

	-- choose a horizontal direction on spawn and keep it for this instance
	local angle = math.rad(math.random(0,359))
	local dir = Vector3.new(math.cos(angle), 0, math.sin(angle)).Unit
	hesitationDir = dir

	local spawnPos = hrp.Position + (dir * 5)

	-- invisible pillar
	local model = Instance.new("Model")
	model.Name = "HesitationModel"

	local pillar = Instance.new("Part")
	pillar.Name = "Pillar"
	pillar.Size = Vector3.new(3,6,3)

	pillar.Anchored = true
	pillar.CanCollide = false

	pillar.Parent = model

	local bb = Instance.new("BillboardGui")
	bb.Name = "EyeBillboard"

	bb.StudsOffset = Vector3.new(0, 0, 0)

	eyeLabel.Name = "EyeLabel"

	eyeLabel.Text = "👁"
	eyeLabel.Font = Enum.Font.GothamBlack
	eyeLabel.TextScaled = true
	eyeLabel.TextColor3 = Color3.fromRGB(160,200,255)
	eyeLabel.TextStrokeTransparency = 0.6
	eyeLabel.ZIndex = 210
	eyeLabel.Parent = bb

	model.Parent = workspace

	hesitationInstance = model
	hesitationSpawnedAt = tick()
	hesitationExposure = 0

	-- flicker screen for 1 second (subtle)
	local flickEnd = tick() + 1
	spawn(function()
		while tick() < flickEnd do
			if rabbitOverlay then
				rabbitOverlay.BackgroundTransparency = 0.78
			end
			wait(0.08)
			if rabbitOverlay then
				rabbitOverlay.BackgroundTransparency = 0.9
			end
			wait(0.06)
		end
		-- ensure final state
		if rabbitOverlay then
			rabbitOverlay.BackgroundTransparency = 0.9
		end
	end)

	-- show hallucination text and exposure UI
	startHallucination()
	exposureGui.Visible = true
	exposureFill.Size = UDim2.new(0,0,1,0)

	-- start exposure accumulation loop (only while the Hesitation pillar is on-screen)
	spawn(function()
		while hesitationInstance and hesitationInstance.Parent do
			wait(hesitationExposureTick)

			-- If the player can't control the camera (cutscene/forced camera), do not accumulate exposure
			if camera and camera.CameraType ~= Enum.CameraType.Custom then
				-- skip this tick; do not increase exposure while player lacks camera control
				continue
			end

			-- ensure the pillar and its billboard still exist and we can test the eye label's world position
			local pillarPart = hesitationInstance:FindFirstChild("Pillar")
			local bb = pillarPart and pillarPart:FindFirstChildOfClass("BillboardGui") or nil
			if not pillarPart or not pillarPart.Parent or not bb then
				-- if the pillar or billboard vanished unexpectedly, break out
				break
			end

			-- compute a world position for the eye label (pillar position plus billboard offset) and test screen presence
			local worldPos = pillarPart.Position + (bb.StudsOffset or Vector3.new(0,0,0))
			local vpPos, onScreen = camera:WorldToViewportPoint(worldPos)
			local inView = onScreen and vpPos.Z > 0 and vpPos.X >= 0 and vpPos.Y >= 0 and vpPos.X <= camera.ViewportSize.X and vpPos.Y <= camera.ViewportSize.Y

			if inView then
				-- exposure rises only while the Hesitation eye label is on-screen
				hesitationExposure = hesitationExposure + hesitationExposureStep
				local pct = math.clamp(hesitationExposure, 0, 100)
				-- update fill using scale relative to parent
				exposureFill.Size = UDim2.new(pct/100,0,1,0)

				-- if reached 100% then kick
				if pct >= 100 then
					local msgs = {
						"Pathetic.",
						"Why did you look?",
						"Curiosity killed the cat.",
						"Falter.",
						"Stupidity is the main trait of a human."
					}
					local msg = msgs[math.random(1,#msgs)]
					local kickText = "DIED TO: HESITATION.\nHOW TO COUNTER: DON'T LOOK.\nMESSAGE: " .. msg
					-- cleanup then kick
					if hesitationInstance and hesitationInstance.Parent then
						hesitationInstance:Destroy()
						hesitationInstance = nil
					end
					stopHallucination()
					-- keep exposure GUI visible (do not hide it on kick)
					task.wait(0.1)
					player:Kick(kickText)
					return
				end
			else
				-- not on screen: do not accumulate exposure (leave current value)
				-- optionally, you could slowly decay exposure here if desired
			end
		end
	end)
end

local function destroyHesitation()
	if hesitationInstance and hesitationInstance.Parent then
		hesitationInstance:Destroy()
		hesitationInstance = nil
	end
	stopHallucination()
	-- keep exposure GUI visible even after Hesitation is destroyed

-- Hesitation spawn loop with progressive percent increase and variable intervals
spawn(function()
	while true do
		-- wait random between 5 and 17 seconds
		local interval = math.random(hesitationMinInterval, hesitationMaxInterval)
		task.wait(interval)
		-- roll chance
		local roll = math.random(1,100)
		if roll <= hesitationChancePercent and not hesitationInstance then
			-- spawn Hesitation
			createHesitation()
			-- Hesitation stays up; schedule auto-despawn after 15-18 seconds
			local activeFor = math.random(hesitationActiveTimeoutMin, hesitationActiveTimeoutMax)
			local startAt = tick()
			spawn(function()
				while hesitationInstance and hesitationInstance.Parent do
					-- if timed out, destroy and reset chance
					if tick() - startAt >= activeFor then
						destroyHesitation()
						hesitationChancePercent = 5
						break
					end
					wait(0.25)
				end
			end)
			-- reset chance after success
			hesitationChancePercent = 5
		else
			-- failure: increase chance by random 5-10%, clamp to 100
			local inc = math.random(hesitationIncreaseMin, hesitationIncreaseMax)
			hesitationChancePercent = math.min(100, hesitationChancePercent + inc)
		end
	end
end)

				rabbitOverlay.BackgroundColor3 = Color3.fromRGB(150,30,30)

			end
		end
	end
end)
